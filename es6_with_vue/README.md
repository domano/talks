### Klassen

```JavaScript
class Multiply {
 
  constructor(factor) {
    this.factor = factor
  }
  
  on(value) {
    return this.factor * value
  }
}

console.log(new Multiply(2).on(21))
```

### Closures
- Neue Syntax: `(a, b) => a+b`
- Sauberes Scoping: Das `this` des Callers bleibt erhalten.

```JavaScript
class Multiply {
 
  constructor(factor) {
    this.factor = factor
  }
  
  onList(list) {
    return list.map((v) => v*this.factor)
  }
}

console.log(
  new Multiply(2)
  .onList([1, 2, 3, 4, 5, 6])
  .reduce((a, b) => a+b))
```

### Modules

```JavaScript
// lib/math.js
export function sum(x, y) {
  return x + y;
}
export var pi = 3.141593;
```

```JavaScript
// app.js
import * as math from "lib/math";
alert("2π = " + math.sum(math.pi, math.pi));
```

```JavaScript
// otherApp.js
import {sum, pi} from "lib/math";
alert("2π = " + sum(pi, pi));
```
